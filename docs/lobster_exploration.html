Find proportion of legal lobsters

lobster_2020 <- lobster_raw |> 
  filter(year == 2020)

legal_lobster <- lobster_2020 |>
  mutate(legal = ifelse(size_mm > 79.76, 
                        yes = "yes", no = "no")) |>
  group_by(site, legal) |> 
  summarize(count = n())
